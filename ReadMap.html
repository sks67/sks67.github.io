def jesusfuckingchrist(r_harray, a):
    nh = np.logspace(np.log10(3.0e17), np.log10(r_harray.max()), r_harray.size)
    n_h_min = 3.0e17
    n_h_max = np.amax(nh)
    norm = (a+1.0)/(n_h_max**(a+1.0) - n_h_min**(a+1.0))
    return norm*(nh**a)

def ccsn_densities(n, bins, r_harray, a, temp_binning):
    lmc_thick = 1.0*3.086e20  #cm                                                                                                                 
    nh = r_harray/lmc_thick
    bins = bins/lmc_thick
    
    n_h_min = 3.0e17/lmc_thick #np.amin(nh)                                                                                                                 
    n_h_max = np.amax(nh)
    
    binvals = bins[1:]-bins[:-1]
    binshift = (np.log10(bins[1:]) - np.log10(bins[:-1]))/2.0
    bincenters = 10**(np.log10(bins[1:])-binshift)
    
    norm = (a+1.0)/(n_h_max**(a+1.0) - n_h_min**(a+1.0))
   # print bincenters.size, n.size
    pdf_nh = norm*(bincenters**a)*n
    #print 'pdf_nh = ',pdf_nh
    norm_ccsn = 1.0/np.sum(pdf_nh*binvals)
    pdf_nh_act = norm_ccsn*pdf_nh
   # print 'pdf_nh_act = ',pdf_nh_act
#Normalizing the discrete pdf and generating random densities                                                                                     
    custm = stats.rv_discrete(name='custm',values=(bincenters,pdf_nh_act*binvals))
    #print pdf_nh_act*binvals
    num_rvs = r_harray.size
    cdf_table = custm.cdf(bincenters)
  #  print cdf_table
    prob_rand = np.random.random(num_rvs)
    if 1:
        plt.plot(bincenters*lmc_thick, cdf_table, 'r.')
        plt.plot(2.0e17*np.ones(prob_rand.size), prob_rand, 'k.')
        plt.xscale('log')
        plt.yscale('log')
        plt.show()
    f = interp.interp1d(cdf_table,bincenters)
    nh_new = f(prob_rand)
    return (nh_new*lmc_thick, bincenters*lmc_thick, norm*(bincenters**a))
